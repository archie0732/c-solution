# 3n+1

[題目連結](https://www.udebug.com/UVa/100)


#### 解題
* 給 a 與 b 要你依據他給的`NP`問題演算法來找到這裏面的最大循環數量

#### code 


```cpp
#include<iostream>
#include<vector>
#include<algorithm>
#include<map>

using namespace std;


int slove(int n, int count)
{
  if (n == 1)
  {
    return count;
  }
  if (n % 2 == 1)
  {
    return slove(3 * n + 1, count + 1);
  }
  else
  {
    return slove(n / 2, count + 1);
  }
}


int main(int argc, char const* argv[])
{
  int a, b;
  while (cin >> a)
  {
    cin >> b;
    cout << a << " " << b << " ";
    int ans1 = 0, ans2 = 0;

    if (a > b)
    {
      int tem = a;
      a = b;
      b = tem;
    }

    for (int i = a;i <= b;i++)
    {
      ans2 = slove(i, 1);
      ans1 = max(ans1, ans2);
    }

    cout << ans1 << endl;

  }

  return 0;
}
```
